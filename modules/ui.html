    Date: 2014-10-27
    Modify: 2014-11-12
    Pen: angelop007@gmail.com

    Example:

    var li1=$('<li>').appendTo($('body>ol#__logs'));
var object_check1=$('<div>').checkUI({
    key:'check1',
    data:[
        {key:'1',val:'A',select:1},
        {key:'2',val:'B'}
    ],
    of:li1
});

var li2=$('<li>').appendTo($('body>ol#__logs'));
var object_check2=$('<div>').checkUI({
    mode:2,
    data:[
        {key:'1',val:'A',select:1},
        {key:'2',val:'B'}
    ],
    of:li2
});
